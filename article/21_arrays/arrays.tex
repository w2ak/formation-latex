        \subsection{Fusion de colonnes}

        \begin{tabular}{|c|l|}
        \hline
        \multicolumn{2}{|c|}{Classement} \\
        \hline
        1 & John Doe \\
        2 & Louis Vaneau \\
        \hline
        \end{tabular}

        \subsection{Fusion de lignes}

        On utilise le package \fbox{\verb!multirow!}

        \begin{tabular}{|c|l|}
        \hline
        \multicolumn{2}{|c|}{Classement} \\
        \hline
        \multirow{3}{*}{1} & John Doe \\
        & Red John \\
        & John Gauss \\
        \hline
        \multirow{2}{*}{2} & Louis Vaneau \\
        \cline{2-2}
        & Louis XVI \\
        \hline
        \end{tabular}
